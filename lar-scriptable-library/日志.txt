def on_scroll(x, y, dx, dy):
    global _wheel
    if dy > 0:
        _wheel = "forward"
    else:
        _wheel = "backward"

# 启动监听 一次性
listener = mouse.Listener(on_scroll=on_scroll)
listener.daemon = True
listener.start()

def check_wheel_change():
    global _wheel
    res = _wheel
    _wheel = None  # 清空，防止重复触发
    return res
